218 /*! \brief [Graph] Creates a Convolutional Network Convolution Layer Node.
219  * \details This function implement Convolutional Network Convolution layer.
220  * For fixed-point data types, a fixed point calculation is performed with round and saturate according to the number of accumulator bits. The number of the accumulator bits are implementation defined,
221  * and should be at least 16.\n
222  * round: rounding according the <tt>vx_round_policy_e</tt> enumeration. \n
223  * saturate: A saturation according the <tt>vx_convert_policy_e</tt> enumeration.
224  * The following equation is implemented: \n
225  * \f$ outputs[j,k,i] = saturate(round(\sum_{l} (\sum_{m,n} inputs[j-m,k-n,l] \times weights[m,n,l,i])+biasses[j,k,i])) \f$\n
226  * Where \f$m,n\f$ are indexes on the convolution matrices. \f$ l\f$ is an index on all the convolutions per input.\f$ i\f$ is an index per output.
227  * \f$ j,k \f$ are the inputs/outputs spatial indexes.
228  * Convolution is done on the width and height dimensions of the <tt>\ref vx_tensor</tt>. Therefore, we use here the term x for index along the width dimension and y for index along the height dimension.\n
229  * before the Convolution is done, a padding with zeros of the width and height input dimensions is performed.
230  * Then down scale is done by picking the results according to a skip jump. The skip in the x and y is determined by the output size dimensions.
231  * The relation between input to output is as follows: \n
232  * \f$ width_{output} = round(\frac{(width_{input} + 2 * padding_x - kernel_x - (kernel_x -1) * dilation_x)}{skip_x} + 1) \f$\n
233  * and \n
234  * \f$ height_{output} = round(\frac{(height + 2 * padding_y - kernel_y - (kernel_y -1) * dilation_y)}{skip_y} + 1) \f$\n
235  * where \f$width\f$ is the size of the input width dimension. \f$height\f$ is the size of the input height dimension.
236  * \f$width_{output}\f$ is the size of the output width dimension. \f$height_{output}\f$ is the size of the output height dimension.
237  * \f$kernel_x\f$ and \f$kernel_y\f$ are the convolution sizes in width and height dimensions.
238  * skip is calculated by the relation between input and output.
239  * rounding is done according to <tt>\ref vx_nn_rounding_type_e</tt>.
240  * \param [in] graph The handle to the graph.
241  * \param [in] inputs The input tensor data. 3 lower dimensions represent a single input, all following dimensions represent number of batches, possibly nested.
242  * The dimension order is [width, height, #IFM, #batches]\n. Implementations must support input tensor data types indicated by the extension strings 'KHR_NN_8' or 'KHR_NN_8 KHR_NN_16'.
243  * \param [in] weights Weights are 4d tensor with dimensions [kernel_x, kernel_y, #IFM, #OFM]. see <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt> \n Weights data type must match the data type of the inputs.
244  * \param [in] biases Optional, ignored if NULL. The biases, which may be shared (one per ofm) or unshared (one per ofm * output location). The possible layouts are
245  * either [#OFM] or [width, height, #OFM]. Biases data type must match the data type of the inputs.
246  * \param [in] convolution_params Pointer to parameters of type <tt>\ref vx_nn_convolution_params_t</tt>
247  * \param [in] size_of_convolution_params Size in bytes of convolution_params.
248  * \param [out] outputs The output tensor data. Output will have the same number and structure of dimensions as input. Output tensor data type must be same as the inputs.
249  * \return <tt> vx_node</tt>.
250  * \returns A node reference <tt>\ref vx_node</tt>. Any possible errors preventing a
251  * successful creation should be checked using <tt>\ref vxGetStatus</tt>.
252  * \ingroup group_cnn
253  */
254 VX_API_ENTRY vx_node VX_API_CALL vxConvolutionLayer(vx_graph graph, vx_tensor inputs, vx_tensor weights, vx_tensor biases, const vx_nn_convolution_params_t *convolution_params, vx_size size_of_convolution_params, vx_tensor outputs);

256 /*! \brief [Graph] Creates a Fully connected Convolutional Network Layer Node.
257  * \details This function implement Fully connected Convolutional Network layers.
258  * For fixed-point data types, a fixed point calculation is performed with round and saturate according to the number of accumulator bits. The number of the accumulator bits are implementation defined,
259  * and should be at least 16.\n
260  * round: rounding according the <tt>vx_round_policy_e</tt> enumeration. \n
261  * saturate: A saturation according the <tt>vx_convert_policy_e</tt> enumeration.
262  * The equation for Fully connected layer:\n
263  * \f$ outputs[i] = saturate(round(\sum_{j} (inputs[j] \times weights[j,i])+biasses[i])) \f$\n
264  * Where \f$j\f$ is a index on the input feature and \f$i\f$ is a index on the output.
265  * before the fully connected is done, a padding of the input is performed.
266  * Then down scale is done by picking the results according to a skip jump. The skip is determined by the output size dimensions.
267  * The relation between input to output is as follows:
268  * \f$ size_{output} = round(\frac{(size_{input} + 2 * padding)}{skip} + 1) \f$\n
269  * where \f$size_{input}\f$ is the size of the input dimension.
270  * \f$size_{output}\f$ is the size of the output dimension.
271  * skip is calculated by the relation between input and output.
272  * rounding is done according to <tt>\ref vx_nn_rounding_type_e</tt>.
273  * \param [in] graph The handle to the graph.
274  * \param [in] inputs The input tensor data. There two possible input layouts:
275  * 1. [#IFM, #batches]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.
276  * 2. [width, height, #IFM, #batches]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>\n
277  * In both cases number of batches are optional and may be multidimensional.
278  * The second option is a special case to deal with convolution layer followed by fully connected.
279  * The dimension order is [#IFM, #batches]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>. Note that batch may be multidimensional. Implementations must support input tensor data types indicated by the extension strings 'KHR_NN_8' or 'KHR_NN_8 KHR_NN_16'.
280  * \param [in] weights Number of dimensions is 2. Dimensions are [#IFM, #OFM]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.\n Implementations must support input tensor data type same as the inputs.
281  * \param [in] biases Optional, ignored if NULL. The biases have one dimension [#OFM]. Implementations must support input tensor data type same as the inputs.
282  * \param [in] overflow_policy A <tt> VX_TYPE_ENUM</tt> of the <tt> vx_convert_policy_e</tt> enumeration.
283  * \param [in] rounding_policy A <tt> VX_TYPE_ENUM</tt> of the <tt> vx_round_policy_e</tt> enumeration.
284  * \param [out] outputs The output tensor data. Output dimension layout is [#OFM,#batches]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>, where #batches may be multidimensional. Output tensor data type must be same as the inputs.
285  * \return <tt> vx_node</tt>.
286  * \returns A node reference <tt>\ref vx_node</tt>. Any possible errors preventing a
287  * successful creation should be checked using <tt>\ref vxGetStatus</tt>.
288  * \ingroup group_cnn
289  */
290 VX_API_ENTRY vx_node VX_API_CALL vxFullyConnectedLayer(vx_graph graph, vx_tensor inputs, vx_tensor weights, vx_tensor biases, vx_enum overflow_policy, vx_enum rounding_policy, vx_tensor outputs);

295  * Pooling operation is a function operation over a rectangle size and then a nearest neighbour down scale.
296  * Here we use pooling_size_x and pooling_size_y to specify the rectangle size on which the operation
297  * is performed. \n
298  * before the operation is done (average or maximum value). the data is padded with zeros in width and height dimensions .

304  * Implementations must support input tensor data types indicated by the extension strings 'KHR_NN_8' or 'KHR_NN_8 KHR_NN_16'.
305  * \param [in] pooling_type Either max pooling or average pooling (see <tt>\ref vx_nn_pooling_type_e</tt>).
306  * \param [in] pooling_size_x Size of the pooling region in the x dimension
307  * \param [in] pooling_size_y Size of the pooling region in the y dimension.
308  * \param [in] pooling_padding_x Padding size in the x dimension.
309  * \param [in] pooling_padding_y Padding size in the y dimension.
310  * \param [in] rounding, Rounding method for calculating output dimensions. See <tt>\ref vx_nn_rounding_type_e</tt>
311  * \param [out] outputs The output tensor data. Output will have the same number of dimensions as input. Output tensor data type must be same as the inputs.
312  * \return <tt> vx_node</tt>.

325 /*! \brief [Graph] Creates a Convolutional Network Softmax Layer Node.
326  * \details the softmax function, is a generalization of the logistic function that "squashes" a K-dimensional vector \f$ z \f$ of arbitrary real values to a K-dimensional vector
327  * \f$ \sigma(z) \f$ of real values in the range (0, 1) that add up to 1. The function is given by:
328  * \f$ \sigma(z) = \frac{\exp^z}{\sum_i \exp^{z_i}} \f$
329  * \param [in] graph The handle to the graph.

332  * In case IFM dimension is 2. Softmax is be calculated on the first dimension. The second dimension is batching.
333  * In case IFM dimension is 3. Dimensions are [Width, Height, Classes]. And Softmax is calculated on the third dimension.
334  * In case IFM dimension is 4. Dimensions are [Width, Height, Classes, batching]. Softmax is calculated on the third dimension.
335  * Regarding the layout specification, see <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.
336  * In all cases Implementations must support input tensor data types indicated by the extension strings 'KHR_NN_8' or 'KHR_NN_8 KHR_NN_16'.
337  * \param [out] outputs The output tensor. Output will have the same number of dimensions as input. Output tensor data type must be same as the inputs.
338  * \ingroup group_cnn
339  * \return <tt> vx_node</tt>.
340  * \returns A node reference <tt>\ref vx_node</tt>. Any possible errors preventing a

346 /*! \brief [Graph] Creates a Convolutional Network Normalization Layer Node. This function is optional for 8-bit extension with the extension string 'KHR_NN_8'.
347  * \details Normalizing over local input regions. Each input value is divided by \f$ (1+\frac{\alpha}{n}\sum_i x^2_i)^\beta \f$ , where n is the number of elements to normalize across.
348  * and the sum is taken over a rectangle region centred at that value (zero padding is added where necessary).
349  * \param [in] graph The handle to the graph.
350  * \param [in] inputs The input tensor data. 3 lower dimensions represent a single input, 4th dimension for batch of inputs is optional.Dimension layout is [width, height, IFM, #batches].
351  * See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.

409 /*! \brief [Graph] Creates a Convolutional Network Deconvolution Layer Node.
410  * \details Deconvolution denote a sort of reverse convolution, which importantly and confusingly is not actually a proper mathematical deconvolution.
411  * Convolutional Network Deconvolution is up-sampling of an image by learned Deconvolution coefficients.
412  * The operation is similar to convolution but can be implemented by up-sampling the inputs with zeros insertions between the inputs,
413  * and convolving the Deconvolution kernels on the up-sampled result.
414  * For fixed-point data types, a fixed point calculation is performed with round and saturate according to the number of accumulator bits. The number of the accumulator bits are implementation defined,
415  * and should be at least 16.\n
416  * round: rounding according the <tt>vx_round_policy_e</tt> enumeration. \n
417  * saturate: A saturation according the <tt>vx_convert_policy_e</tt> enumeration.
418  * The following equation is implemented: \n
419  * \f$ outputs[j,k,i] = saturate(round(\sum_{l} \sum_{m,n}(inputs_{upscaled}[j-m,k-n,l] \times weights[m,n,l,i])+biasses[j,k,i])) \f$\n
420  * Where \f$m,n\f$ are indexes on the convolution matrices. \f$ l\f$ is an index on all the convolutions per input.\f$ i\f$ is an index per output.
421  * \f$ j,k \f$ are the inputs/outputs spatial indexes.
422  * Deconvolution is done on the width and height dimensions of the <tt>\ref vx_tensor</tt>. Therefore, we use here the term x for the width dimension and y for the height dimension.\n
423  * before the Deconvolution is done, up-scaling the width and height dimensions with zeros is performed.
424  * The relation between input to output is as follows: \n
425  * \f$ width_{output} = round( (width_{input} -1) * upscale_x - 2 * padding_x + kernel_x + a_x) \f$\n
426  * and \n
427  * \f$ height_{output} = round( (height_{input} - 1) * upscale_y - 2 * padding_y + kernel_y + a_y) \f$\n
428  * where \f$width_{input}\f$ is the size of the input width dimension. \f$height_{input}\f$ is the size of the input height dimension.
429  * \f$width_{output}\f$ is the size of the output width dimension. \f$height_{output}\f$ is the size of the output height dimension.
430  * \f$kernel_x\f$ and \f$kernel_y\f$ are the convolution sizes in width and height. \f$a_x\f$ and \f$a_y\f$ are user-specified quantity used to distinguish between the \f$upscale_x\f$ and \f$upscale_y\f$ different possible output sizes
431  * \f$upscale_x\f$ and \f$upscale_y\f$ are calculated by the relation between input and output.
432  * rounding is done according to <tt>\ref vx_nn_rounding_type_e</tt>.
433  * \param [in] graph The handle to the graph.
434  * \param [in] inputs The input tensor. 3 lower dimensions represent a single input, and an optional 4th dimension for batch of inputs. Dimension layout is [width, height, #IFM, #batches].
435  * See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.
436  * Implementations must support input tensor data types indicated by the extension strings 'KHR_NN_8' or 'KHR_NN_8 KHR_NN_16'.
437  * \param [in] weights The 4d weights with dimensions [width, height, #IFM, #OFM]. See <tt>\ref vxCreateTensor</tt> and <tt>\ref vxCreateVirtualTensor</tt>.
438  * \param [in] biases Optional, ignored if NULL. The biases have one dimension [#OFM]. Implementations must support input tensor data type same as the inputs.
439  * \param [in] deconvolution_params Pointer to parameters of type <tt>\ref vx_nn_deconvolution_params_t</tt>
440  * \param [in] size_of_deconv_params Size in bytes of deconvolution_params.
441  * \param [out] outputs The output tensor. The output has the same number of dimensions as the input.
442  * \ingroup group_cnn
443  * \return <tt> vx_node</tt>.
444  * \returns A node reference <tt>\ref vx_node</tt>. Any possible errors preventing a
445  * successful creation should be checked using <tt>\ref vxGetStatus</tt>.
446  */
447 VX_API_ENTRY vx_node VX_API_CALL vxDeconvolutionLayer(vx_graph graph, vx_tensor inputs, vx_tensor weights, vx_tensor biases, const vx_nn_deconvolution_params_t *deconvolution_params, vx_size size_of_deconv_params, vx_tensor outputs);
